from flask import Flask, render_template, request, redirect
from flask_sqlalchemy import SQLAlchemy
from flask_login import LoginManager, UserMixin, login_user, login_required, logout_user, current_user
import pdfplumber

app = Flask(_name_)
app.config['SECRET_KEY'] = 'supersecretkey'
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///database.db'
db = SQLAlchemy(app)

login_manager = LoginManager()
login_manager.init_app(app)
login_manager.login_view = "login"

# ---------------- Models ---------------- #

class User(UserMixin, db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(100), unique=True)
    password = db.Column(db.String(100))
    role = db.Column(db.String(20))  # admin / user


class Voter(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(200))
    father = db.Column(db.String(200))
    para = db.Column(db.String(200))


@login_manager.user_loader
def load_user(user_id):
    return User.query.get(int(user_id))


# ------------- Routes ------------- #

@app.route("/", methods=["GET", "POST"])
def login():
    if request.method == "POST":
        user = User.query.filter_by(
            username=request.form["username"],
            password=request.form["password"]
        ).first()

        if user:
            login_user(user)
            if user.role == "admin":
                return redirect("/admin")
            else:
                return redirect("/search")

    return render_template("login.html")


@app.route("/admin", methods=["GET", "POST"])
@login_required
def admin():
    if current_user.role != "admin":
        return "Unauthorized"

    if request.method == "POST":
        file = request.files["pdf"]
        if file:
            with pdfplumber.open(file) as pdf:
                for page in pdf.pages:
                    text = page.extract_text()
                    if text:
                        lines = text.split("\n")
                        for line in lines:
                            parts = line.split(",")
                            if len(parts) >= 3:
                                voter = Voter(
                                    name=parts[0].strip(),
                                    father=parts[1].strip(),
                                    para=parts[2].strip()
                                )
                                db.session.add(voter)

                db.session.commit()

    return render_template("admin.html")


@app.route("/search", methods=["GET", "POST"])
@login_required
def search():
    results = []

    if request.method == "POST":
        name = request.form["name"]
        father = request.form["father"]
        para = request.form["para"]

        query = Voter.query

        if name:
            query = query.filter(Voter.name.contains(name))
        if father:
            query = query.filter(Voter.father.contains(father))
        if para:
            query = query.filter(Voter.para.contains(para))

        results = query.all()

    return render_template("search.html", results=results)


@app.route("/logout")
def logout():
    logout_user()
    return redirect("/")


# --------- Create Default Users --------- #

@app.before_first_request
def create_tables():
    db.create_all()

    if not User.query.filter_by(username="admin").first():
        admin = User(username="admin", password="1234", role="admin")
        user = User(username="user", password="1234", role="user")

        db.session.add(admin)
        db.session.add(user)
        db.session.commit()


if _name_ == "_main_":
    app.run()
